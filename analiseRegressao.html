modelo1 = lm(insatisfacao ~ normalizados, data = reclamacoes_l)

tidy(modelo1, conf.int = TRUE, conf.level = 0.95)
glance(modelo1)
reclamacoes_augmented = reclamacoes_l %>% 
  add_predictions(modelo1)
